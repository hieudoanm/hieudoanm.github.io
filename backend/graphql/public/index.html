chess

Response

Returns a Chess

Example

Query
query chess {
  chess {
    countries {
      ...TitledCountriesFragment
    }
    country {
      ...TitledCountryFragment
    }
    openings {
      ...OpeningFragment
    }
    player {
      ...PlayerFragment
    }
    streamers {
      ...StreamersFragment
    }
    titled {
      ...TitledFragment
    }
  }
}

    "chess": {
      "countries": [TitledCountries],
      "country": TitledCountry,
      "openings": [Opening],
      "player": Player,
      "streamers": Streamers,
      "titled": Titled
    }
  }
}

Mutations

games

Response

Returns a GamesSynced

Arguments
Name Description
username - String!
month - Int
year - Int

Example

Query
mutation games(
  $username: String!,
  $month: Int,
  $year: Int
) {
  games(
    username: $username,
    month: $month,
    year: $year
  ) {
    total
    synced
    existed
  }
}
Variables
{
  "username": "abc123",
  "month": 123,
  "year": 987
}
Response
{"data": {"games": {"total": 987, "synced": 123, "existed": 987}}}
